```jsx live
function Counter() {
  const [count, setCount] = useState(0)

  return (
    <>
      <p>count: {count}</p>
      <div style={{ display: 'flex', gap: 10 }}>
        <button onClick={() => setCount(count + 1)}>increase</button>
        <button onClick={() => setCount(count - 1)}>decrease</button>
      </div>
    </>
  )
}
```
